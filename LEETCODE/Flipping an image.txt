class Solution {
    public int[][] flipAndInvertImage(int[][] image) { 
         int row = image.length;
        int col = image[0].length;
        for (int i = 0; i < row; i++) {
            for (int j = 0, m=col-1; j <= m; j++,m--) {
                int temp = image[i][j]^1;
                image[i][j] = image[i][m]^1;
                image[i][m] = temp;
            }
        }
        return image;  
    }
}
